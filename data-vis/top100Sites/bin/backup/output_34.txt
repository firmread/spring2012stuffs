	<script type="text/javascript">
		document.write('<style type="text/css">#counter-container { visibility: hidden }</style>');

		var iPhoneOSVersion = AC.Detector.isMobile() ? AC.Detector.iOSVersion() : null,
			isCountdownShowing = false,
			downloadCounter = new DownloadCounter("/autopush/us/itunes/includes/countdown.inc"),
			maxCount = 24999999990,
			setup = function() {
				var currentCount = downloadCounter.currentCount(),
					counterPromo = $('counter-promo'),
					counterPromoAlt = $('counter-promo-alt');
				if (currentCount >= maxCount) {
					counterPromo.hide();
					downloadCounter.stop();
					counterPromoAlt.show();
				} else {
					counterPromoAlt.hide();
					counterPromo.hide();
					downloadCounter.setElement($('downloadCounter'));
				}
				$('counter-container').style.visibility = 'visible';
			};

		downloadCounter.setMaxCount(maxCount);
		downloadCounter.setDigitImageAnimationCount(6);

		
		if ((iPhoneOSVersion && iPhoneOSVersion[0] < 4)) {
			downloadCounter.setDigitImageHeight(145);
			document.write('<style type="text/css">#downloadCounter .digit { height:145px; }</style>');
		} else {	
			downloadCounter.setDigitImageHeight(145);
		}
		
		downloadCounter.setDelegate({
			counterDidReachValue: function(downloadCounter, currentCount) {
				if(currentCount >= maxCount) {
					downloadCounter.stop();
					$('counter-promo').remove();
					new Effect.Appear('counter-promo-alt', {duration:0.8});
				} else {
					if(!isCountdownShowing && !isNaN(downloadCounter.currentCount())){
						$('counter-promo').show();
						isCountdownShowing = true;
					}
				}
			},
			didLoadData: function() {
				Event.onDOMReady(setup.bind(this));
			}
		});
	</script>
