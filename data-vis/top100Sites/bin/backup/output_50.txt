      function params(wut,qp,dflt){ dflt=(dflt==null)?'':dflt; try{r=unescape(wut.match(new RegExp(qp+"=+([^&;]*)"))[1]);}catch(qp){r=dflt;} return r; }

      
      function getZip() 
      {
        var zip = "";        
        r = params(document.cookie,'Zip','null');         // get zip var from cookie
        
        if( r != 'null' )
        {            
          zip = r.toString();
          return zip;
        } 
        else 
        {            
          r = params(location.search,'Zip','');           // get zip var from url
          zip = r.toString();
        }

        return zip;
      }

      
      // setZipCookie - sets the zip cookie if it is different than what already exists
      function setZipCookie(zipcode) 
      {
        var cookiename = "Zip";
        var existingzip = params(document.cookie,'Zip',null);
        if ( existingzip == null || existingzip != zipcode ) 
        {
          var zip = zipcode;
          var MINUTE = 60 * 1000;
          var HOUR = MINUTE * 60;
          var DAY = HOUR * 24;
          var WEEK = DAY * 7;
          var now = (new Date()).getTime();
          var futureMS = WEEK * 4;
          var futureDate = new Date(now + futureMS);
          var expdate = futureDate.toGMTString();
          if ( zipcode == null ) 
          {
            var pastDate = new Date(now - DAY);
            expdate = pastDate.toGMTString();
          }
          document.cookie = cookiename+"="+zip+"; expires="+expdate+"; path=/";
          return true;
        } 
        else { return true; }
      }

      {
        var regexp = /\s*(\S*(\s*\S)*)\s*/;
